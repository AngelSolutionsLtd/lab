export const IconGallery = () => {
  const [search, setSearch] = useState("");
  const [copied, setCopied] = useState(null);
  const [copyFormat, setCopyFormat] = useState("class");
  const [iconSize, setIconSize] = useState(28);
  const [iconColor, setIconColor] = useState("#111827");
  const inputRef = useRef(null);

  useEffect(() => {
    const handleKey = (e) => {
      if (e.key === "Escape") {
        setSearch("");
        inputRef.current?.blur();
      }
      if (e.key === "/" && document.activeElement !== inputRef.current) {
        e.preventDefault();
        inputRef.current?.focus();
      }
    };
    window.addEventListener("keydown", handleKey);
    return () => window.removeEventListener("keydown", handleKey);
  }, []);

  const filtered = icons.filter(name =>
    name.includes(search.toLowerCase().trim())
  );

  const handleCopy = (name) => {
    const text = copyFormat === "class"
      ? `entypo--${name}`
      : `<span class="entypo--${name}"></span>`;
    navigator.clipboard.writeText(text);
    setCopied(name);
    setTimeout(() => setCopied(null), 1500);
  };

  return (
    <div>
      <div style={{
        display: "flex",
        flexWrap: "wrap",
        alignItems: "center",
        gap: "12px",
        marginBottom: "24px",
        padding: "12px 16px",
        backgroundColor: "#f9fafb",
        borderRadius: "8px",
        border: "1px solid #e5e7eb",
      }}>

        {/* Search */}
        <div style={{ position: "relative" }}>
          <input
            ref={inputRef}
            type="text"
            placeholder='Search icons…'
            value={search}
            onChange={e => setSearch(e.target.value)}
            style={{
              padding: "7px 30px 7px 12px",
              fontSize: "14px",
              border: "1px solid #d1d5db",
              borderRadius: "6px",
              width: "260px",
              outline: "none",
              backgroundColor: "#fff",
            }}
          />
          {search && (
            <button
              onClick={() => { setSearch(""); inputRef.current?.focus(); }}
              style={{
                position: "absolute",
                right: "8px",
                top: "50%",
                transform: "translateY(-50%)",
                background: "none",
                border: "none",
                cursor: "pointer",
                color: "#9ca3af",
                fontSize: "18px",
                lineHeight: 1,
                padding: 0,
              }}
            >
              ×
            </button>
          )}
        </div>

        {/* Divider */}
        <div style={{ width: "1px", height: "28px", backgroundColor: "#d1d5db" }} />

        {/* Copy format toggle */}
        <div style={{ display: "flex", alignItems: "center", gap: "6px" }}>
          <span style={{ fontSize: "12px", color: "#6b7280" }}>Copy as</span>
          <div style={{ display: "flex", border: "1px solid #d1d5db", borderRadius: "6px", overflow: "hidden" }}>
            {[
              { value: "class", label: ".class" },
              { value: "html",  label: "<span>" },
            ].map(({ value, label }) => (
              <button
                key={value}
                onClick={() => setCopyFormat(value)}
                style={{
                  padding: "5px 10px",
                  fontSize: "12px",
                  border: "none",
                  borderRight: value === "class" ? "1px solid #d1d5db" : "none",
                  cursor: "pointer",
                  backgroundColor: copyFormat === value ? "#6366f1" : "#fff",
                  color: copyFormat === value ? "#fff" : "#374151",
                  fontFamily: "monospace",
                  transition: "background-color 0.15s",
                }}
              >
                {label}
              </button>
            ))}
          </div>
        </div>

        {/* Divider */}
        <div style={{ width: "1px", height: "28px", backgroundColor: "#d1d5db" }} />

        {/* Size slider */}
        <div style={{ display: "flex", alignItems: "center", gap: "8px" }}>
          <span style={{ fontSize: "12px", color: "#6b7280" }}>Size</span>
          <input
            type="range"
            min="16"
            max="64"
            value={iconSize}
            onChange={e => setIconSize(Number(e.target.value))}
            style={{ width: "80px", cursor: "pointer" }}
          />
          <span style={{ fontSize: "12px", color: "#6b7280", width: "28px" }}>
            {iconSize}px
          </span>
        </div>

        {/* Divider */}
        <div style={{ width: "1px", height: "28px", backgroundColor: "#d1d5db" }} />

        {/* Colour picker */}
        <div style={{ display: "flex", alignItems: "center", gap: "8px" }}>
          <span style={{ fontSize: "12px", color: "#6b7280" }}>Colour</span>
          <input
            type="color"
            value={iconColor}
            onChange={e => setIconColor(e.target.value)}
            style={{
              width: "30px",
              height: "30px",
              border: "1px solid #d1d5db",
              borderRadius: "4px",
              cursor: "pointer",
              padding: "2px",
              backgroundColor: "#fff",
            }}
          />
          <button
            onClick={() => setIconColor("#111827")}
            style={{
              fontSize: "11px",
              color: "#9ca3af",
              background: "none",
              border: "none",
              cursor: "pointer",
              padding: 0,
              textDecoration: "underline",
            }}
          >
            Reset
          </button>
        </div>

        {/* Count — pushed to the right */}
        <span style={{ marginLeft: "auto", fontSize: "12px", color: "#6b7280", whiteSpace: "nowrap" }}>
          {filtered.length} of {icons.length} icons
        </span>
      </div>

      {filtered.length === 0 ? (
        <p style={{ color: "#6b7280", fontStyle: "italic" }}>
          No icons match "{search}"
        </p>
      ) : (
        <div style={{
          display: "grid",
          gridTemplateColumns: "repeat(auto-fill, 130px)",
          justifyContent: "center",
          gap: "12px",
        }}>
          {filtered.map(name => {
            const isCopied = copied === name;
            return (
              <div
                key={name}
                onClick={() => handleCopy(name)}
                title={`Click to copy`}
                style={{
                  display: "flex",
                  flexDirection: "column",
                  alignItems: "center",
                  justifyContent: "center",
                  gap: "10px",
                  padding: "16px 8px",
                  border: `1px solid ${isCopied ? "#6366f1" : "#e5e7eb"}`,
                  borderRadius: "8px",
                  cursor: "pointer",
                  backgroundColor: isCopied ? "#eef2ff" : "#fff",
                  transition: "border-color 0.15s, background-color 0.15s",
                  minHeight: "90px",
                }}
                onMouseEnter={e => {
                  if (!isCopied) {
                    e.currentTarget.style.borderColor = "#9ca3af";
                    e.currentTarget.style.backgroundColor = "#f9fafb";
                  }
                }}
                onMouseLeave={e => {
                  if (!isCopied) {
                    e.currentTarget.style.borderColor = "#e5e7eb";
                    e.currentTarget.style.backgroundColor = "#fff";
                  }
                }}
              >
                <span
                  className={`entypo--${name}`}
                  style={{
                    fontSize: `${iconSize}px`,
                    color: isCopied ? "#6366f1" : iconColor,
                    transition: "font-size 0.1s",
                  }}
                />
                <span style={{
                  fontSize: "10px",
                  color: isCopied ? "#6366f1" : "#6b7280",
                  textAlign: "center",
                  wordBreak: "break-all",
                  lineHeight: "1.3",
                  fontFamily: "monospace",
                }}>
                  {isCopied ? "Copied!" : name}
                </span>
              </div>
            );
          })}
        </div>
      )}
    </div>
  );
};
